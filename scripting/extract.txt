SELECT
  C.ACCOUNT_CODE AS CREDIT_CODE,
  C.CLIENT_CODE,
  C.CLIENT_DOCN,
  C.CLIENT_NAME,
  C.PORTFOLIO_NAME,
  C.PRINCIPAL_AMNT,
  C.NET_DEBT_AMNT,
  N.NEGOTIATOR_CODE,
  N.NEGOTIATOR_NAME
FROM DATA.DIM_CREDIT_1 C
INNER JOIN DATA.DIM_NEGOTIATOR N ON N.NEGOTIATOR_NAME = C.NEGOTIATOR_NAME
WHERE C.FLAG = 1;
